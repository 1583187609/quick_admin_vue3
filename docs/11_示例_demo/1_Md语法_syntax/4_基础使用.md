## 国际化

[详见](https://vitepress.dev/zh/guide/i18n)

```ts
import { defineConfig } from "vitepress";

export default defineConfig({
  // 共享属性和其他顶层内容...

  locales: {
    root: {
      label: "English",
      lang: "en",
    },
    fr: {
      label: "French",
      lang: "fr", // 可选，将作为 `lang` 属性添加到 `html` 标签中
      link: "/fr/guide", // 默认 /fr/ -- 显示在导航栏翻译菜单上，可以是外部的

      // 其余 locale 特定属性...
    },
  },
});
```
